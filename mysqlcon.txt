import mysql.connector as c
con=c.connect(host="localhost",user="root",passwd="mysql", database="student")
cursor=con.cursor()
while True:
    id=int(input("Enter Student ID: "))
    name=input("Enter Student Name: ")
    addr=input("Enter Student Addr: ")
    query="Insert into student_info values({},'{}','{}')".format(id,name,addr)
    cursor.execute(query)
    con.commit()
    print("Data Inserted Successfully...")
    x=int(input("1:Enter More\n2:Exit\nEnter choice: "))
    if x==2:
        break
id1=input("Enter Student ID you want to Update: ")
name1=input("Enter Student New Name: ")
query="update student_info set name='{}' where ID={}".format(name1,id1)
cursor.execute(query)

print("Data Updated Successfully...")
